### Runtime

```javascript
// On element selection
function onElementSelected(element) {
    // Get element classes
    const classes = parser.getElementClasses(element);
    
    // Check for undefined classes
    const undefined = parser.getUndefinedClasses(element, availableClasses);
    
    // Render UI
    renderClassTags(classes, undefined);
    renderStylesPreview(classes);
}

// On class add
function addClass(className) {
    // Validate
    if (!parser.classExists(className, cssText)) {
        showWarning('Class not defined');
    }
    
    // Add to element
    element.classList.add(className);
    
    // Update UI
    onElementSelected(element);
}
```
